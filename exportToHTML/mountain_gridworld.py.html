18   class MountainGridWorld():
19       """ 
20       The world is a 5 x 5 grid based on Example 3.5 from Sutton 2019. There are 25 states. We index these states as follows: 
21    
22           0   1   2   3   4 
23           5   6   7   8   9 
24           10  11  12  13  14 
25           15  16  17  18  19 
26           20  21  22  23  24 
27    
28       For example, state "1" is cell "A" in Sutton 2019, state "3" is cell "B", and so forth. 
29    
30       There are 4 actions. We index these actions as follows: 
31    
32                   1 (up) 
33           2 (left)        0 (right) 
34                   3 (down) 
35    
36       If you specify hard_version=True, then the action will be selected uniformly at random 10% of the time. 
37       """
38   
39       def __init__(self, hard_version=False):
40           self.num_states = 49
41           self.num_actions = 4
42           self.last_action = None
43           self.max_num_steps = 1000
44           self.state_shape = (7, 7)
45           self.state_names = ['x', 'y']
46           self.start_state = 8
47           self.end_state = 37
48           self.reset()
49   
50       def altitude(self, s):
51           # s must be one integer smaller than 7*7
52           assert s < self.num_states
53   
54           x, y = self.get_pos(s)
55           altitude = altitude_lut[x][y]
56           assert altitude >= 0
57   
58       def get_pos(self, s):
59           assert self.state_shape[0] == self.state_shape[1]
60           return s // self.state_shape[0], s % self.state_shape[0]
61   
62       def p(self, s1, s, a):
63           raise NotImplementedError
64   
65       def r(self, s):
66           if s == self.end_state:
67               return 100  # goal reward
68           elif s < 0 or s >= 49:
69               return -10
70           else:
71               return -1  # time penalty
72   
73       def step(self, a, verbose=False):
74           last_s = self.s
75   
76           # real s1
77           s1 = self.apply_action_with_uncertainty(self.s, a)
78           r = self.r(s1)
79           done = self.is_done(s1)
80           # print("from state", self.s, "action", a, "to state", s1, "reward", r, "done", done)
81   
82           if done:
83               s1 = self.s
84               obs = s1
85               confidence = 1
86           else:
87               # noisy obs
88               obs, confidence = self.get_obs(s1)
89               self.s = s1
90           self.num_steps += 1
91   
92           if verbose:
93               print(f"from state {last_s} action {a} to state {s1} reward {r} done {done} confidence {confidence}")
94   
95           # if done:
96           #     print(f"done at {self.num_steps} steps")
97           return s1, r, done, obs, confidence

99       def get_next_obs_and_confidence(self, s1, obs, a):
100          s1_obs = self.apply_action(obs, a)
101          confidence = self.get_confidence(s1)
102          return s1_obs, confidence

104      def get_confidence(self, s):
105          x, y = self.get_pos(s)
106          altitude = altitude_lut[x][y]
107          return 0.5 + 0.09 * altitude
108  
109      def is_done(self, s):
110          return s == self.end_state or self.num_steps >= self.max_num_steps or s < 0 or s >= 49
111  
112      def get_possible_actions(self, s, a=None):
113          # if s is in border
114          x, y = self.get_pos(s)
115          actions = None
116          if y == 0:
117              if x == 0:
118                  actions = [RIGHT, DOWN]
119              elif x == 6:
120                  actions = [RIGHT, UP]
121              else:
122                  actions = [RIGHT, UP, DOWN]
123          elif y == 6:
124              if x == 0:
125                  actions = [LEFT, DOWN]
126              elif x == 6:
127                  actions = [LEFT, UP]
128              else:
129                  actions = [LEFT, UP, DOWN]
130          elif x == 0:
131              actions = [RIGHT, LEFT, DOWN]
132          elif x == 6:
133              actions = [RIGHT, LEFT, UP]
134          else:
135              actions = [RIGHT, LEFT, UP, DOWN]
136  
137          assert actions is not None, f"actions is None for state {s}, {x}, {y}"
138          if a is not None and a in actions:
139              # remove a from actions
140              actions.remove(a)
141  
142          return actions

144      @staticmethod
145      def apply_action(s, a):
146          if a == RIGHT:
147              return s + 1
148          if a == UP:
149              return s - 7
150          if a == LEFT:
151              return s - 1
152          if a == DOWN:
153              return s + 7
154          return s
155  
156      def apply_action_with_uncertainty(self, s, a):
157          accurate_p = 0.75
158          possible_actions = self.get_possible_actions(s)
159          # if move is impossible, move the state to -1 for penalty
160          if a not in possible_actions:
161              return -1
162          possible_actions.remove(a)
163          if random.uniform(0, 1) < accurate_p:
164              # accurate
165              return self.apply_action(s, a)
166          else:
167              random_action = random.choice(self.get_possible_actions(s, a))
168              return self.apply_action(s, random_action)
169  
170      def get_possible_wrong_obs(self, s):
171          possible_actions = self.get_possible_actions(s)
172          possible_s = []
173          for a in possible_actions:
174              obs = self.apply_action(s, a)
175              assert 0 <= obs < 49, f"Invalid state {s}, {a}, {obs}, {self.get_pos(s)}"
176              possible_s.append(obs)
177          return possible_s
178  
179      def get_possible_wrong_r(self, s):
180          possible_wrong_s = self.get_possible_wrong_obs(s)
181          possible_wrong_r = []
182          for s in possible_wrong_s:
183              possible_wrong_r.append(self.r(s))
184          return possible_wrong_r
185  
186      def get_obs(self, s):
187          x, y = self.get_pos(s)
188          assert self.state_shape[0] > x >= 0 and self.state_shape[1] > y >= 0, f"Invalid state {s}, {x}, {y}"
189          altitude = altitude_lut[x][y]
190          if altitude < 0:
191              return s, 1
192  
193          obs_p = 0.5 + 0.09 * altitude
194          obs = random.uniform(0, 1)
195          if obs < obs_p:
196              return s, obs_p
197          else:
198              possible_s = self.get_possible_wrong_obs(s)
199              # TODO: check if choice is uniform
200              return random.choice(possible_s), obs_p
201  
202      def reset(self):
203          # Choose initial state uniformly at random
204          self.s = self.start_state
205          self.num_steps = 0
206          self.last_action = None
207          return self.s
